\newcommand{\myFigure}[5]{
    %% example:
    %       \myFigure{} %% filename in figures folder
    %       {width=0.5\textwidth,height=0.5\textheight} %% maximum width/height, aspect ratio will be kept
    %       {} %% caption
    %       {} %% optional (short) caption for list of figures
    %       {} %% label
    \begin{figure}[htbp]
    \centering
    \includegraphics[keepaspectratio,#2]{#1}
    \caption[#4]{#3}
    \label{fig:#5}
    \end{figure}
}


\newcommand{\myDoubleFigure}[9]{
    %% example:
    % \myDoubleFigure{} %% filename1 in figures folder
    %       {width=0.5\textwidth,height=0.5\textheight} %% maximum width/height, aspect ratio will be kept
    %       {} %% label 1
    %       {} %% filename2 in figures folder
    %       {width=0.5\textwidth,height=0.5\textheight} %% maximum width/height, aspect ratio will be kept
    %       {} %% label 2
    %       {} %% caption
    %       {} %% optional (short) caption for list of figures
    %       {} %% label
    \begin{figure}[htbp]%
        \centering
        \subfloat[#3]{
            \label{fig:#9a} 
            \includegraphics[#2]{#1} 
        }%
        \hspace{.25cm}
        \subfloat[#6]{
            \label{fig:#9b} 
            \includegraphics[#5]{#4} 
        }%
        \caption[#8]{#7}%
        \label{fig:#9}%
    \end{figure}
}
